**Result:** Work transferred and workspace cleaned up automatically


## Complete Workflow (One Function)

**Copy-paste ready:**

```python
def complete_validated_workflow(task_description, project_type):
    """
    Complete workflow with validation and analysis.
    Time: ~32 minutes (vs 77 minutes without v3.3)
    """
    # 1. Generate validated prompt
    task_prompt = generate_validation_prompt(project_type, task_description)
    
    # 2. Create feature branch
    feature_branch = f"feature/{task_description.lower().replace(' ', '-')[:30]}-{int(time.time())}"
    coder_workspace_bash(
        workspace=home_workspace,
        command=f"cd {repo_path} && git checkout -b {feature_branch} && git push -u origin {feature_branch}",
        timeout_ms=30000
    )
    
    # 3. Create task
    task = coder_create_task(
        input=task_prompt,
        template_version_id=template_id,
        rich_parameter_values={
            "feature_branch": feature_branch,
            "home_workspace": home_workspace
        }
    )
    
    # 4. Wait for completion
    # ... (monitor task status until idle)
    
    # 5. Transfer work
    complete_task_with_cleanup(
        task_workspace=task.workspace_name,
        home_workspace=home_workspace,
        feature_branch=feature_branch,
        task_description=task_description
    )
    
    # 6. Run analysis
    analysis = complete_post_task_analysis(home_workspace, repo_path)
    
    # 7. Quality gates
    from validation_patterns import pre_merge_quality_gate, pre_deployment_quality_gate
    
    if not pre_merge_quality_gate(home_workspace, repo_path, project_type):
        print("❌ Pre-merge gate failed")
        return False
    
    if not pre_deployment_quality_gate(home_workspace, repo_path):
        print("❌ Pre-deployment gate failed")
        return False
    
    print("✅ Ready for deployment!")
    return True

# Usage
success = complete_validated_workflow(
    task_description="Implement user authentication",
    project_type="python"
)
```

## Quality Gates

### Pre-Merge Gate

```python
from validation_patterns import pre_merge_quality_gate

if pre_merge_quality_gate(home_workspace, repo_path, project_type):
    # Safe to merge
    merge_feature_branch()
else:
    # Fix issues first
    print("Quality gate failed - address issues")
```

### Pre-Deployment Gate

```python
from validation_patterns import pre_deployment_quality_gate

if pre_deployment_quality_gate(home_workspace, repo_path, min_consistency=85, min_compliance=90):
    # Safe to deploy
    deploy_to_production()
else:
    # Not ready
    print("Deployment blocked - review analysis reports")
```

```python
# 1. Create validated task
task_prompt = generate_validation_prompt('python', 'Add user profile feature')
task = coder_create_task(input=task_prompt, ...)

# 2. Wait and transfer
complete_task_with_cleanup(...)

# 3. Analyze
analysis = complete_post_task_analysis(home_workspace, repo_path)

# 4. Quality gate

    merge_and_deploy()
```

### Pattern 2: Bug Fix

```python
# 1. Create validated task with bug context
task_prompt = generate_validation_prompt('nodejs', 'Fix login timeout bug')
task = coder_create_task(input=task_prompt, ...)

# 2. Wait and transfer
complete_task_with_cleanup(...)

# 3. Verify fix
validation_results = verify_task_completion(home_workspace, repo_path, 'nodejs')
```

### Pattern 3: Multiple Parallel Tasks

```python
# 1. Create multiple validated tasks
tasks = []

    task_prompt = generate_validation_prompt('python', desc)

# 2. Wait for all to complete
# ... (monitor all tasks)

# 3. Transfer all work

    complete_task_with_cleanup(...)

analysis = complete_post_task_analysis(home_workspace, repo_path)
```

## Troubleshooting

### Issue: Analysis takes too long

**Solution:** Limit scope to critical deliverables
```python
# Analyze only specific files
deliverables = {
    'DESIGN.md': {...},
    'IMPLEMENTATION_PLAN.md': {...}
}
```

### Issue: Validation false positives

**Solution:** Adjust quality gate thresholds
```python
# Lower thresholds if needed
pre_deployment_quality_gate(
    home_workspace, 
    repo_path,
    min_consistency=75,  # Default: 85
    min_compliance=80    # Default: 90
)
```

### Issue: Reports not generated

**Solution:** Check file paths and permissions
```python
# Verify repo path is correct
result = coder_workspace_bash(
    workspace=home_workspace,
    command=f"ls -la {repo_path}",
    timeout_ms=5000
)
print(result.stdout)
```
